date 12 December 2019

authors Dawid Huczyński,

Door numbers

I’ve needed to have custom door numbers in spreadsheet. Simple script that will assign room number depending on the floor. From bottom left to top right. In such scheme: [floor number][room number]
Example: fifth room on first floor would have 105.

One door family for all door swings

At my office we were using different door families for different door swings. Common was to have door type with name ending like IN-900-left, IN-800-right. It was the same doors but different names. Moreover, you had to change the family if you changed the door swing. So many error posibilities.

This script added new shared parameter: swing. From now you could run the script to update the swing parameter from doors geometry, and print spreadsheet with doors :). Easy.

"""Rename doors:
    type marks: [ED-01: exterior door type 01] by function, 
                family and family type

    marks: [2-05: 2nd level fifth door on the floor]
"""
import clr

clr.AddReference("RevitAPI")
from Autodesk.Revit.DB import (
    FilteredElementCollector,
    BuiltInCategory,
    Transaction,
    TransactionGroup,
    XYZ,
)

clr.AddReference("RevitServices")
from RevitServices.Persistence import DocumentManager

# Grab current document
doc = DocumentManager.Instance.CurrentDBDocument


def get_elements_by_category(category, doc):
    elements = (
        FilteredElementCollector(doc)
        .OfCategory(category)
        .WhereElementIsNotElementType()
        .ToElements()
    )
    return elements


def number_elements(elements, doc):
    # check if elements are correct
    assert elements
    for el in elements:
        assert el.Category.Name in ["Windows", "Doors", "Walls"]
    el_type = el.Category.Name
    # Sorts doors by key
    elements_by_key = {}
    if el_type == 'Walls':
        key = lambda x: x.Width
    else:
        key = lambda x: x.Symbol.LookupParameter("Width")
    for el in sorted(elements, key=key):
        family = el.Symbol.FamilyName
        typ = el.Name
        if el_type == "Doors":
            function = el.Symbol.LookupParameter("Function").AsValueString()
        elif el_type == "Windows":
            function = "Window"
        # Could be other parameters (if exists !!)
        # fire_rating = el.Symbol.LookupParameter("Fire Rating").AsString()
        key = function + family + typ
        if elements_by_key.get(key):
            elements_by_key[key] = elements_by_key[key] + [el]
        else:
            elements_by_key[key] = [el]

    # Assign mark id to each door separated by key and rising width
    tg = TransactionGroup(doc, "{} number assing".format(el_type))
    tg.Start()
    Interior = 0
    Exterior = 0
    Curtain = 0
    Windows = 0
    Walls = 0  # TODO
    key = lambda x: x[1][0].Symbol.LookupParameter("Width")
    OUT = []
    for k, items in sorted(elements_by_key.items(), key=key):
        if el_type == "Windows":
            Windows += 1
            nr = Windows
            name = "O"
        elif k.startswith("Interior"):
            Interior += 1
            nr = Interior
            name = "DW" if el_type == "Doors" else "SW"
        elif k.startswith("Exterior"):
            Exterior += 1
            nr = Exterior
            name = "DZ" if el_type == "Doors" else "SZ"
        else:
            Curtain += 1
            nr = Curtain
            name = items[0].Symbol.Name

        t = Transaction(doc, "Change {} type marks".format(el_type))
        t.Start()
        # TODO: Add swing parameter if doesnt exists
        # TODO: update swing parameter by flip state
        # swing = "L" if door.GetParameterValueByName("swing") == "Left" else "P"
        items[0].Symbol.LookupParameter("Type Mark").Set(
            "{}-{:02d}{}".format(name, nr, "")
        )
        t.Commit()

    # Assign individual number for each door from left to right [level]-[nr]
    floors = {}
    for el in elements:
        if el.Symbol.FamilyName.startswith("Underground"):
            # elements to ommit
            continue
        # format next line depends on your document language
        # "Poziom 1" is example level value
        level = el.LookupParameter("Level").AsValueString().split(" ")[1]
        if floors.get(level):
            floors[level] = floors[level] + [el]
        else:
            floors[level] = [el]

    for floor, elements in floors.items():
        key = lambda x: x.Location.Point.DistanceTo(XYZ(-100000, -10000, 0))
        sorted_elements = sorted(elements, key=key)
        t = Transaction(doc, "Change {} marks".format(el_type))
        t.Start()
        for nr, el in enumerate(sorted_elements):
            el.LookupParameter("Mark").Set("{}-{:02d}".format(floor, nr + 1))
        t.Commit()

    tg.Assimilate()
    return floors


# Get elements from document
all_doors = get_elements_by_category(BuiltInCategory.OST_Doors, doc)
all_windows = get_elements_by_category(BuiltInCategory.OST_Windows, doc)
all_walls = get_elements_by_category(BuiltInCategory.OST_Walls, doc)


doors = number_elements(all_doors, doc)
windows = number_elements(all_windows, doc)
# TODO: Walls
# walls = number_elements(all_walls, doc)
walls =[]

# Assign your output to the OUT variable.
OUT = {"windows": windows, "doors": doors, "walls": walls}
